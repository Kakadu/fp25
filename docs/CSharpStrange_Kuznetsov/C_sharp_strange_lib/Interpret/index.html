type interpret_error =
  1. | NotImplemented
  2. | NoVariable of string
  3. | AddressNotFound of int
  4. | VarDeclared of string
  5. | TypeMismatch
  6. | ImpossibleResult of string
  7. | OtherError of string
type error =
  1. | IError of interpret_error
val pp_error : Stdlib.Format.formatter -> error -> unit
val show_error : error -> string
type adr =
  1. | Adr of int
type value =
  1. | VInt of int
  2. | VBool of bool
  3. | VChar of char
  4. | VString of string
  5. | VNull
  6. | VObject of adr
val pp_value : Stdlib.Format.formatter -> value -> unit
val show_value : value -> string
val interpret_program : Ast.program -> (value option, interpret_error) Stdlib.result
val interpret : string -> (value option, interpret_error) Stdlib.result
