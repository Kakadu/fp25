type binop =
  1. | Add
    (*

    e1 + e2 сложение целых

    *)
  2. | Sub
    (*

    e1 - e2 вычитание

    *)
  3. | Mul
    (*

    e1 * e2 умножение

    *)
  4. | Div
    (*

    e1 / e2 целочисленное деление, ошибка при делении на 0

    *)
  5. | Eq
    (*

    e1 = e2 равно: результат 1, если равны, иначе 0

    *)
  6. | Lt
    (*

    e1 < e2 меньше: результат 1, если e1 < e2, иначе 0

    *)
  7. | Gt
    (*

    e1 > e2 больше: результат 1, если e1 > e2, иначе 0

    *)

type 'name t =
  1. | Var of 'name
    (*

    Переменная x. Примеры: x, n, fact.

    *)
  2. | Abs of 'name * 'name t
    (*

    Анонимная функция fun x -> body

    *)
  3. | App of 'name t * 'name t
    (*

    Применение функции f a

    *)
  4. | Int of int
    (*

    Целое число. Примеры: 0, 1, 42, -5.

    *)
  5. | Let of 'name * 'name t * 'name t
    (*

    Нерекурсивное определение: let x = e1 in e2 Сначала вычисляем e1 , результат связываем с именем x, потом вычисляем e2 в окружении

    *)
  6. | Let_rec of 'name * 'name * 'name t * 'name t
    (*

    Рекурсивное определение функции: let rec f x = body in e

    *)
  7. | If of 'name t * 'name t * 'name t
    (*

    If (cond, e1, e2) условное выражение: если cond не равно нулю вычислить e1, иначе e2

    *)
  8. | Binop of binop * 'name t * 'name t
    (*

    Binop (add, sub, div) бинарная операция над двумя выражениями

    *)
  9. | Fix of 'name t
    (*

    Оператор фиксированной точки fix f

    *)
