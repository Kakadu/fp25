type name = string
type rec_flag =
  1. | NonRecursive
    (*

    let x = e1 in e2

    *)
  2. | Recursive
    (*

    let rec f = e1 in e2

    *)

Recursion flag for let bindings

type binop =
  1. | Plus
    (*

    Integer addition: +

    *)
  2. | Minus
    (*

    Integer subtraction: -

    *)
  3. | Mul
    (*

    Integer multiplication: *

    *)
  4. | Div
    (*

    Integer division: /

    *)

Binary operations

type compop =
  1. | Equal
    (*

    Equal: =

    *)
  2. | NotEqual
    (*

    Not equal: <>

    *)
  3. | Less
    (*

    Less than: <

    *)
  4. | LessEq
    (*

    Less or equal: <=

    *)
  5. | Greater
    (*

    Greater than: >

    *)
  6. | GreaterEq
    (*

    Greater or equal: >=

    *)

Comparison operations

type constant =
  1. | Int of int
    (*

    Integer literal

    *)
  2. | Unit
    (*

    Unit value

    *)

Constant values

type unop =
  1. | Neg
    (*

    Unary minus: -e

    *)

Unary operations

type expr =
  1. | Const of constant
    (*

    Constant value

    *)
  2. | Var of name
    (*

    Variable reference

    *)
  3. | Abs of name list * expr
    (*

    Multi-argument abstraction: fun x y -> e

    *)
  4. | App of expr * expr
    (*

    Function application: f arg

    *)
  5. | Let of rec_flag * name * expr * expr
    (*

    Let binding: let[rec] x = e1 in e2

    *)
  6. | BinOp of binop * expr * expr
    (*

    Binary arithmetic operation

    *)
  7. | UnOp of unop * expr
    (*

    Unary operation

    *)
  8. | Comp of compop * expr * expr
    (*

    Comparison operation

    *)
  9. | If of expr * expr * expr
    (*

    Conditional: if cond then e1 else e2

    *)
