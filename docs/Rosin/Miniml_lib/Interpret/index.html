Module Miniml_lib.Interpret

type value =
  1. | VUnit
  2. | VNum of int
  3. | VClosure of Ast.varname * Ast.expr * env
  4. | VRecClosure of Ast.varname * Ast.varname * Ast.expr * env
and env = (Ast.varname * value) list
type error =
  1. | UnboundVariable of Ast.varname
  2. | DivisionByZero
  3. | StepLimitExceeded
  4. | NonFunctionApplication of value
  5. | NonIntegerCondition of value
  6. | InvalidBinop of Ast.binop * value * value
val run_interpret : Ast.expr -> int -> (value, error) Stdlib.Result.t
