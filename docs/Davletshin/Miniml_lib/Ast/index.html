binary operators

type name = string
type 'name t =
  1. | Int of int
    (*

    Integer literal

    *)
  2. | Var of 'name
    (*

    Variable x

    *)
  3. | Abs of 'name * 'name t
    (*

    Abstraction fun x -> t

    *)
  4. | App of 'name t * 'name t
    (*

    Application f g

    *)
  5. | Binop of bop * 'name t * 'name t
    (*

    Binary operator a op b

    *)
  6. | Unop of uop * 'name t
    (*

    Unary operator op e

    *)
  7. | If of 'name t * 'name t * 'name t
    (*

    Condition if c then t else e

    *)
  8. | Let of flag * 'name * 'name t * 'name t
    (*

    Let binding let [rec] p = e1 in e2

    *)
  9. | Fix of 'name t
    (*

    Fix-point combinator fix (fun self -> e)

    *)
  10. | Print of 'name t
    (*

    Built-in print. Only integers print n

    *)

The main type for our abstract syntax tree
