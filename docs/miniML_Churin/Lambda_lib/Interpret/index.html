type error =
  1. | UnboundVariable of Ast.var
  2. | NotAnInteger of value
  3. | NotAFunction of value
  4. | DivisionByZero
  5. | StepLimitExceeded
  6. | InvalidCondition of value
  7. | FixAppliedToNonFunction
and value =
  1. | VInt of int
  2. | VUnit
  3. | VClosure of Ast.var * Ast.expr * env
  4. | VRecClosure of Ast.var * Ast.var * Ast.expr * env
  5. | VBuiltin of value -> (value, error) Stdlib.result
and env = (Ast.var * value) list
module InterpM : sig ... end

val initial_env : env

val apply : value -> value -> value InterpM.t
val default_steps : int
