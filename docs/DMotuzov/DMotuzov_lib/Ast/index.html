type expression =
  1. | Expr_var of identificator
    (*

    Variable, e.g. `x`

    *)
  2. | Expr_const of constant
    (*

    Constant, e.g. integer or unit

    *)
  3. | Expr_binary_op of binary_op * expression * expression
    (*

    Binary operation, e.g. `e1 + e2`

    *)
  4. | Expr_conditional of expression * expression * expression
    (*

    Conditional expression, e.g. `if e1 then e2 else e3`

    *)
  5. | Expr_let_in of identificator * expression * expression
    (*

    Local binding, e.g. `let x = e1 in e2`

    *)
  6. | Expr_let_rec_in of identificator * expression * expression
    (*

    Local rec binding, e.g. `let rec x = e1 in e2`

    *)
  7. | Expr_fun of identificator * expression
    (*

    Abstraction (function), e.g. `fun x -> e`

    *)
  8. | Expr_ap of expression * expression list
    (*

    Function application, e.g. `f x`

    *)
  9. | Expr_fix of expression
    (*

    Fixed-point combinator for recursion, e.g. `fix (fun f x -> ...)` Enables defining recursive functions without explicit `let rec`.

    *)
