type typ =
  1. | TInt
  2. | TFun of typ * typ
  3. | TUnit
  4. | TVar of string
type type_env = (string, typ, Base.String.comparator_witness) Base.Map.t
type type_error =
  1. | UnboundVar of string
  2. | TypeMismatch of typ * typ
  3. | OccursCheckError
  4. | InvalidCondition
  5. | ApplicationError
  6. | NotExpression
val typecheck_program : Ast.expr -> (typ, type_error) Stdlib.result
