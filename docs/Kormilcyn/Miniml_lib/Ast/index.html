Module Miniml_lib.Ast

type name = string
type binop =
  1. | Add
    (*

    Integer addition

    *)
  2. | Mul
    (*

    Integer multiplication

    *)
  3. | Sub
    (*

    Integer subtraction

    *)
  4. | Div
    (*

    Integer division

    *)
  5. | Leq
    (*

    Integer less-or-equal comparison

    *)
  6. | Eq
    (*

    Integer equality comparison

    *)
  7. | Geq
    (*

    Integer greater-or-equal comparison

    *)
type 'name t =
  1. | Var of 'name
    (*

    Variable x

    *)
  2. | Fun of 'name * 'name t
    (*

    Function fun x -> ...

    *)
  3. | App of 'name t * 'name t
    (*

    Application f g

    *)
  4. | Int of int
    (*

    Integer literal

    *)
  5. | Neg of 'name t
    (*

    Unary minus

    *)
  6. | Bin of binop * 'name t * 'name t
    (*

    Binary operator application

    *)
  7. | Let of 'name * 'name t * 'name t
    (*

    Let binding

    *)
  8. | If of 'name t * 'name t * 'name t
    (*

    Conditional expression

    *)
  9. | LetRec of 'name * 'name t * 'name t
    (*

    Recursive let binding

    *)
  10. | Fix
    (*

    Fixed-point operator

    *)

The main type for our AST (дерева абстрактного синтаксиса)
