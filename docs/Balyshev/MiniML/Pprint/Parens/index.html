Module Pprint.Parens

type context =
  1. | Free
    (*

    top-level let and expressions after in are free

    *)
  2. | Binop
    (*

    XXX + XXX * (XXX - XXX)

    *)
  3. | Tuple
    (*

    XXX, XXX

    *)
  4. | App
    (*

    XXX XXX

    *)
  5. | LeftSideFun
    (*

    fun XXX -> ...

    *)
  6. | RightSideFun
    (*

    fun ... -> XXX

    *)
  7. | MatchWith
    (*

    match XXX with ...

    *)
  8. | LeftSideMatch
    (*

    match ... with | XXX -> ...

    *)
  9. | RightSideMatch
    (*

    match ... with | ... -> XXX

    *)
  10. | LeftSideArrow
    (*

    'XXX -> ... in type declaration and infered types

    *)
  11. | RightSideArrow
    (*

    ... -> 'XXX in type declaration and infered types

    *)
  12. | LeftSideLet
    (*

    let XXX = ... in ...

    *)
  13. | RightSideLet
    (*

    let ... = XXX in ...

    *)
  14. | Ite
    (*

    if XXX then XXX else XXX

    *)
val parens : string -> string
val set_parens : ctx:'a -> 'a Base.List.t -> string -> string
