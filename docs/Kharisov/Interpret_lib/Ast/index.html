type expr =
  1. | EConst of int
    (*

    integer literal: 42, -1

    *)
  2. | EVar of id
    (*

    variable: x, fact

    *)
  3. | EBinop of binop * expr * expr
    (*

    binary operation: e1 + e2

    *)
  4. | EIf of expr * expr * expr
    (*

    conditional: if c then e1 else e2

    *)
  5. | EFun of id * expr
    (*

    abstraction: fun x -> e

    *)
  6. | EApp of expr * expr
    (*

    application: f x

    *)
  7. | ELet of rec_flag * id * expr * expr
    (*

    binding: let (rec)? x = e1 in e2

    *)
