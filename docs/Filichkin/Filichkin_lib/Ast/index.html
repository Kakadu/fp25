Module Filichkin_lib.Ast

type binop =
  1. | Plus
    (*

    Addition: `+`

    *)
  2. | Minus
    (*

    Subtraction: `-`

    *)
  3. | Mult
    (*

    Multiplication: `*`

    *)
  4. | Div
    (*

    Division: `/`

    *)
  5. | Equal
    (*

    Equality: `=`

    *)
  6. | More
    (*

    Greater than: `>`

    *)
  7. | Less
    (*

    Less than: `<`

    *)
  8. | EMore
    (*

    Greater than or equal: `>=`

    *)
  9. | ELess
    (*

    Less than or equal: `<=`

    *)
type rec_flag =
  1. | NonRec
    (*

    Non-recursive binding

    *)
  2. | Rec
    (*

    Recursive binding

    *)
type expr =
  1. | Int of int
    (*

    Integer literal

    *)
  2. | Var of string
    (*

    Variable name

    *)
  3. | BinOp of binop * expr * expr
    (*

    Binary operation: operator, left and right operands

    *)
  4. | If of expr * expr * expr option
    (*

    Conditional expression: condition, then branch, optional else branch

    *)
  5. | Let of rec_flag * string * expr * expr option
    (*

    Let expression: recursiveness flag, name, bound expression, body (optional) The last parameter is the optional expression body (for parsing convenience)

    *)
  6. | Abs of expr * expr
    (*

    Abstraction (lambda function): parameter list, function body Syntactic sugar for functions with multiple arguments

    *)
  7. | App of expr * expr
    (*

    Application (function call): function, argument

    *)
