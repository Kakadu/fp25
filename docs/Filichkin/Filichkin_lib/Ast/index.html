type binop =
  1. | Plus
    (*

    Addition: `+`

    *)
  2. | Minus
    (*

    Subtraction: `-`

    *)
  3. | Mult
    (*

    Multiplication: `*`

    *)
  4. | Div
    (*

    Division: `/`

    *)
  5. | Equal
    (*

    Equality: `=`

    *)
  6. | NotEqual
    (*

    Not Equal `<>`

    *)
  7. | More
    (*

    Greater than: `>`

    *)
  8. | Less
    (*

    Less than: `<`

    *)
  9. | EMore
    (*

    Greater than or equal: `>=`

    *)
  10. | ELess
    (*

    Less than or equal: `<=`

    *)
  11. | And
    (*

    `&&`

    *)
  12. | Or
    (*

    `||`

    *)
val pp_binop : Ppx_deriving_runtime.Format.formatter -> binop -> Ppx_deriving_runtime.unit
val show_binop : binop -> Ppx_deriving_runtime.string
type rec_flag =
  1. | NonRec
    (*

    Non-recursive binding

    *)
  2. | Rec
    (*

    Recursive binding

    *)
val pp_rec_flag : Ppx_deriving_runtime.Format.formatter -> rec_flag -> Ppx_deriving_runtime.unit
val show_rec_flag : rec_flag -> Ppx_deriving_runtime.string
type unop =
  1. | Neg
    (*

    Unary negation

    *)
  2. | Not
    (*

    Logical NOT

    *)
val pp_unop : Ppx_deriving_runtime.Format.formatter -> unop -> Ppx_deriving_runtime.unit
val show_unop : unop -> Ppx_deriving_runtime.string
type ident = string
val pp_ident : Ppx_deriving_runtime.Format.formatter -> ident -> Ppx_deriving_runtime.unit
val show_ident : ident -> Ppx_deriving_runtime.string
type type_expr =
  1. | TEInt
    (*

    Integer type

    *)
  2. | TEBool
    (*

    Boolean type

    *)
  3. | TEUnit
    (*

    Unit type

    *)
  4. | TEVar of ident
    (*

    Type variable

    *)
  5. | TEArrow of type_expr * type_expr
    (*

    Function type

    *)
  6. | TETuple of type_expr list
    (*

    Tuple type

    *)
  7. | TEConstr of ident * type_expr list
    (*

    Type constructor application

    *)
val pp_type_expr : Ppx_deriving_runtime.Format.formatter -> type_expr -> Ppx_deriving_runtime.unit
val show_type_expr : type_expr -> Ppx_deriving_runtime.string
type constr_decl = {
  1. ctor_name : ident;
  2. ctor_args : type_expr list;
}
type type_decl = {
  1. type_name : ident;
  2. type_params : ident list;
  3. constructors : constr_decl list;
}
type pattern =
  1. | PVar of ident
    (*

    Variable pattern

    *)
  2. | PTuple of pattern list
    (*

    Tuple pattern

    *)
  3. | PConstr of ident * pattern list
    (*

    Constructor pattern

    *)
  4. | PWildcard
    (*

    Wildcard pattern

    *)
val pp_pattern : Ppx_deriving_runtime.Format.formatter -> pattern -> Ppx_deriving_runtime.unit
val show_pattern : pattern -> Ppx_deriving_runtime.string
type expr =
  1. | Int of int
    (*

    Integer literal

    *)
  2. | Var of ident
    (*

    Variable name

    *)
  3. | Constr of ident
    (*

    Constr name

    *)
  4. | BinOp of binop * expr * expr
    (*

    Binary operation: operator, left and right operands

    *)
  5. | Bool of bool
    (*

    Boolean literal

    *)
  6. | UnOp of unop * expr
    (*

    Unary operation: operator, operand

    *)
  7. | If of expr * expr * expr
    (*

    Conditional expression: condition, then branch, optional else branch

    *)
  8. | Let of rec_flag * pattern * expr * expr option
    (*

    Let expression: recursiveness flag, name, bound expression, body (optional) The last parameter is the optional expression body (for parsing convenience)

    *)
  9. | Abs of pattern * expr
    (*

    Abstraction (lambda function): parameter, Syntactic sugar for functions with multiple arguments

    *)
  10. | App of expr * expr
    (*

    Application (function call): function, argument

    *)
  11. | Tuple of expr list
    (*

    Tuple expression

    *)
  12. | Match of expr * (pattern * expr) list
    (*

    Pattern matching

    *)
val pp_expr : Ppx_deriving_runtime.Format.formatter -> expr -> Ppx_deriving_runtime.unit
val show_expr : expr -> Ppx_deriving_runtime.string
type toplevel =
  1. | TLExpr of expr
    (*

    Top-level expression

    *)
  2. | TLType of type_decl
    (*

    Top-level type declaration

    *)
val pp_toplevel : Ppx_deriving_runtime.Format.formatter -> toplevel -> Ppx_deriving_runtime.unit
val show_toplevel : toplevel -> Ppx_deriving_runtime.string
