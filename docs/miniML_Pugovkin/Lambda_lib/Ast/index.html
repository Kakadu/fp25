type expr =
  1. | Const of const

    Литерал

    *)

    Идентификатор

    *)
  3. | Lam of ident * expr
    (*

    Лямбда-абстракция: fun x -> e

    *)
  4. | App of expr * expr
    (*

    Аппликация: e1 e2

    *)
  5. | Let of rec_flag * ident * expr * expr
    (*

    Связывание: let x = e1 in e2

    *)

    *)

    Унарная операция

    *)

    Арифметика

    *)

    (*

    Сравнение

    *)
  10. | BinopBool of bool_binops * expr * expr
    (*

    Логические операции

    *)
  11. | Fix of expr
    (*

    Фиксация: fix e
